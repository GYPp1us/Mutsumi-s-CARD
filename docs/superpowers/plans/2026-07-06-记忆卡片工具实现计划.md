# 记忆卡片工具 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建一个原生 Android 记忆卡片 MVP，支持文字 key、图片 value、触控笔录入、轻量随机推荐、响应式布局和手动备份。

**Architecture:** 使用单 Activity + Jetpack Compose Navigation。核心业务按 `study`、`cards`、`draw`、`backup`、`data`、`ui/adaptive` 分包，数据通过 Repository 暴露 `Flow` 状态，图片保存在应用私有目录，Room 只保存结构化数据和相对路径。

**Tech Stack:** Kotlin、Jetpack Compose、Material 3、Room、DataStore、Kotlin Serialization、JUnit、Robolectric、AndroidX Test。

package com.mutsumi.card.domain.review

import kotlin.math.max
import kotlin.math.min

class ReviewWeightPolicy {
    fun nextWeight(currentWeight: Double, feedback: ReviewFeedback): Double {
        val next = when (feedback) {
            ReviewFeedback.Again -> min(currentWeight + 0.8, 5.0)
            ReviewFeedback.Unsure -> min(currentWeight + 0.3, 5.0)
            ReviewFeedback.Know -> max(currentWeight * 0.55, 0.2)
        }
        return ((next * 100.0).toInt()) / 100.0
    }
}

Create `app/src/main/java/com/mutsumi/card/domain/review/WeightedCardPicker.kt`:

```kotlin
package com.mutsumi.card.domain.review

import kotlin.random.Random

class WeightedCardPicker(
    private val random: Random = Random.Default,
) {
    fun pick(
        cards: List<WeightedReviewCard>,
        recentCardIds: List<Long>,
    ): WeightedReviewCard? {
        if (cards.isEmpty()) return null
        val recentSet = recentCardIds.takeLast(3).toSet()
        val candidates = if (cards.size > recentSet.size + 1) {
            cards.filterNot { it.cardId in recentSet }
        } else {
            cards
        }
        val total = candidates.sumOf { it.weight.coerceAtLeast(0.01) }
        var cursor = random.nextDouble(total)
        for (candidate in candidates) {
            cursor -= candidate.weight.coerceAtLeast(0.01)
            if (cursor <= 0.0) return candidate
        }
        return candidates.last()
    }
}
```

class ImageResourceStore(private val context: Context) {

        val file = File(imageDir, fileName)
        file.writeBytes(bytes)
        return "images/${file.name}"
    }

    fun readImage(relativePath: String): ByteArray {
        return File(context.filesDir, relativePath).readBytes()
    }

    fun copyFromUri(uri: Uri, fileName: String): String {
        val bytes = context.contentResolver.openInputStream(uri)?.use { it.readBytes() }
            ?: error("无法读取图片")
        return writeImage(fileName, bytes)
    }

package com.mutsumi.card.backup

import kotlinx.serialization.encodeToString
import kotlinx.serialization.json.Json
import java.io.OutputStream
import java.util.zip.ZipEntry
import java.util.zip.ZipOutputStream

class BackupService(
    private val json: Json = Json { prettyPrint = true },
) {
    fun export(
        snapshot: BackupSnapshot,
        images: Map<String, ByteArray>,
        output: OutputStream,
        exportedAt: Long,
    ) {
        ZipOutputStream(output).use { zip ->
            val manifest = BackupManifest(
                version = 1,
                exportedAt = exportedAt,
                appVersion = "0.1.0",
                deckCount = snapshot.decks.size,
                cardCount = snapshot.cards.size,
            )
            zip.writeTextEntry("manifest.json", json.encodeToString(manifest))
            zip.writeTextEntry("database.json", json.encodeToString(snapshot))
            images.forEach { (path, bytes) ->
                zip.putNextEntry(ZipEntry(path))
                zip.write(bytes)
                zip.closeEntry()
            }
        }
    }

    private fun ZipOutputStream.writeTextEntry(name: String, text: String) {
        putNextEntry(ZipEntry(name))
        write(text.toByteArray(Charsets.UTF_8))
        closeEntry()
    }
}

package com.mutsumi.card.data.repository

import com.mutsumi.card.core.time.Clock
import com.mutsumi.card.data.local.CardDao
import com.mutsumi.card.data.local.DeckDao
import com.mutsumi.card.data.local.ReviewDao
import com.mutsumi.card.data.local.entity.CardEntity
import com.mutsumi.card.data.local.entity.DeckEntity
import com.mutsumi.card.data.local.entity.ReviewStateEntity
import com.mutsumi.card.domain.review.ReviewFeedback
import com.mutsumi.card.domain.review.ReviewWeightPolicy

class CardRepository(
    private val deckDao: DeckDao,
    private val cardDao: CardDao,
    private val reviewDao: ReviewDao,
    private val clock: Clock,
    private val weightPolicy: ReviewWeightPolicy = ReviewWeightPolicy(),
) {
    suspend fun createDeck(name: String): Long {
        val now = clock.nowMillis()
        return deckDao.insertDeck(DeckEntity(name = name, createdAt = now, updatedAt = now))
    }

    suspend fun createCard(deckId: Long, keyText: String, valueImagePath: String, baseImagePath: String?): Long {
        val now = clock.nowMillis()
        val cardId = cardDao.insertCard(
            CardEntity(
                deckId = deckId,
                keyText = keyText,
                valueImagePath = valueImagePath,
                baseImagePath = baseImagePath,
                createdAt = now,
                updatedAt = now,
            ),
        )
        reviewDao.upsertReviewState(ReviewStateEntity(cardId = cardId))
        return cardId
    }

    suspend fun applyFeedback(cardId: Long, feedback: ReviewFeedback) {
        val current = reviewDao.getReviewState(cardId) ?: ReviewStateEntity(cardId = cardId)
        val nextWeight = weightPolicy.nextWeight(current.weight, feedback)
        val updated = when (feedback) {
            ReviewFeedback.Again -> current.copy(
                weight = nextWeight,
                seenCount = current.seenCount + 1,
                againCount = current.againCount + 1,
                lastReviewedAt = clock.nowMillis(),
            )
            ReviewFeedback.Unsure -> current.copy(
                weight = nextWeight,
                seenCount = current.seenCount + 1,
                unsureCount = current.unsureCount + 1,
                lastReviewedAt = clock.nowMillis(),
            )
            ReviewFeedback.Know -> current.copy(
                weight = nextWeight,
                seenCount = current.seenCount + 1,
                knownCount = current.knownCount + 1,
                lastReviewedAt = clock.nowMillis(),
            )
        }
        reviewDao.upsertReviewState(updated)
    }
}

    val strokes: List<DrawingStroke> = emptyList(),
)

class DrawViewModel : ViewModel() {
    private val _uiState = MutableStateFlow(DrawUiState())
    val uiState: StateFlow<DrawUiState> = _uiState

    fun updateKey(text: String) {
        _uiState.update { it.copy(keyText = text) }
    }

    fun addStroke(stroke: DrawingStroke) {
        _uiState.update { it.copy(strokes = it.strokes + stroke) }
    }

    fun undo() {
        _uiState.update { it.copy(strokes = it.strokes.dropLast(1)) }
    }

    fun clear() {
        _uiState.update { it.copy(strokes = emptyList()) }
    }
}

@Composable
fun DrawScreen(
    state: DrawUiState,
    onKeyChange: (String) -> Unit,
    onStroke: (DrawingStroke) -> Unit,
    onUndo: () -> Unit,
    onClear: () -> Unit,
) {
    val currentPoints = remember { mutableStateListOf<StrokePoint>() }
    Column(modifier = Modifier.fillMaxSize(), verticalArrangement = Arrangement.spacedBy(12.dp)) {
        OutlinedTextField(
            value = state.keyText,
            onValueChange = onKeyChange,
            label = { Text("文字 key") },
            modifier = Modifier.fillMaxWidth(),
        )
        Canvas(
            modifier = Modifier
                .fillMaxWidth()
                .aspectRatio(4f / 3f)
                .background(Color(0xFFF8FAF9), RoundedCornerShape(8.dp))
                .padding(8.dp)
                .pointerInput(state.penColor, state.penWidth) {
                    detectDragGestures(
                        onDragStart = { currentPoints.clear(); currentPoints += StrokePoint(it) },
                        onDrag = { change, _ -> currentPoints += StrokePoint(change.position) },
                        onDragEnd = {
                            onStroke(DrawingStroke(currentPoints.toList(), state.penColor, state.penWidth))
                            currentPoints.clear()
                        },
                    )
                },
        ) {
            fun drawStroke(stroke: DrawingStroke) {
                if (stroke.points.size < 2) return
                val path = Path().apply {
                    moveTo(stroke.points.first().position.x, stroke.points.first().position.y)
                    stroke.points.drop(1).forEach { lineTo(it.position.x, it.position.y) }
                }
                drawPath(path, stroke.color, style = Stroke(width = stroke.width, cap = StrokeCap.Round))
            }
            state.strokes.forEach(::drawStroke)
            drawStroke(DrawingStroke(currentPoints.toList(), state.penColor, state.penWidth))
        }
        Row(horizontalArrangement = Arrangement.spacedBy(8.dp), modifier = Modifier.fillMaxWidth()) {
            OutlinedButton(onClick = onUndo, modifier = Modifier.weight(1f)) { Text("撤销") }
            OutlinedButton(onClick = onClear, modifier = Modifier.weight(1f)) { Text("清空") }
            Button(onClick = {}, modifier = Modifier.weight(1f)) { Text("保存") }
        }
    }
}

name: Release APK

on:
  push:
    tags:
      - "v*"
  workflow_dispatch:
    inputs:
      tag_name:
        description: "Release tag, for example v0.1.0"
        required: true
        type: string

permissions:
  contents: write

jobs:
  release:
    name: Build and publish APK
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v7.0.0

      - name: Set up JDK
        uses: actions/setup-java@v5.4.0
        with:
          distribution: temurin
          java-version: "17"

      - name: Set up Gradle
        uses: gradle/actions/setup-gradle@v6.2.0

      - name: Make Gradle executable
        run: chmod +x ./gradlew

      - name: Build debug APK
        run: ./gradlew :app:assembleDebug

      - name: Prepare release asset
        run: |
          mkdir -p release
          cp app/build/outputs/apk/debug/app-debug.apk release/mutsumi-card-debug.apk

      - name: Resolve release tag
        id: release_tag
        shell: bash
        run: |
          if [[ "${{ github.event_name }}" == "workflow_dispatch" ]]; then
            echo "tag=${{ inputs.tag_name }}" >> "$GITHUB_OUTPUT"
          else
            echo "tag=${GITHUB_REF_NAME}" >> "$GITHUB_OUTPUT"
          fi

      - name: Publish GitHub release
        uses: softprops/action-gh-release@v3
        with:
          tag_name: ${{ steps.release_tag.outputs.tag }}
          name: 记忆卡片 ${{ steps.release_tag.outputs.tag }}
          files: release/mutsumi-card-debug.apk
          generate_release_notes: true
